 46     function turn() {
 47         select(false);
 48         $('.legal').removeClass('legal'); // clears legal moves of last moved piece
 49         $('.active').removeClass('active'); // Clear active status of previous players pieces << B. Fisher 5/6 1700
 50         // Find whether the last move placed the next player in check
 51         Game.Players[1].King.inCheck = (check(Game.Players[1].King.position, Game.Players[0]));
 52         //Clears the EP variables of the current players pawns << B. Fisher
 53         $(Game.Players[0].pawns).each(function() {
 54             this.EP = false;
 55         });
 56         Game.Players.reverse(); // Switches the active player
 57         Game.Players[0].activate();
 58         if (Game.Players[0].color == 'white') { //Each time both players have moved turnCount incriments <<B. Fisher
 59             Game.turnCount++;
 60         }
 61         Game.change = null;
 62         if (Checkmate()) endGame(2);
 63         if (Stalemate()) endGame(3);
 64         // Game didn't end, change name and color of dash
 65         $('#Dash').css('background', Game.Players[0].color);
 66         $('#turn').html(Game.Players[0].name);
 67         $('.threat').removeClass('threat');
 68     }

329 	pawn.prototype.footprint = function() {
330 		self = this;
331         row = self.row();
332         col = self.col();
333         ids = [];
334         if (!self.moved) ids.push(col + (row + 2 * this.inc));
335         else ids.push(col + (row + 1 * this.inc));
336         ids.push(cLabels[cLabels.indexOf(col) - 1] + (row + this.inc));
337         ids.push(cLabels[cLabels.indexOf(col) + 1] + (row + this.inc));
338         return ids;
339 	};
340     pawn.prototype.move = function(destination) {
341 		self = this;
342         // Evaluate whether pawn has passed an opponent pawn on opening double step << B. Fisher
343         if (!this.moved && (destination.id.match(4) || destination.id.match(5))) {
344             var prev = (this.col() == 'A') ? false : occupied(destination.previousElementSibling.id);
345             var next = (this.col() == 'H') ? false : occupied(destination.nextElementSibling.id);
346             if (prev && prev == 'pawn' && prev.color != self.color) prev.EP = this;
347             if (next && next == 'pawn' && next.color != self.color) next.EP = this;
348         }
349         /** Check if pawn is capturing en passant. EP variable will hold an opponent pawn that passed. If opponent pawn and this pawn are not on the same column (position[0]) clear EP variable. << B. Fisher */
350         if (this.EP && this.EP.position[0] != destination.id[0]) this.EP = false;
351         this._move(destination);
352         if (this.row() == this.endRow) this.promote(destination);
353 	};

538     function Legal(piece) {
539         var footprint = piece.footprint(),
540             color = piece.color,
541             type = piece.type,
542             position = piece.position,
543             rNum = position[1] * 1,
544             C = position[0],
545             cNum = cLabels.indexOf(C) + 1,
546             dest, orig, path, legalIDs = "";
547         var legal = new Array();
548         //	console.log(color + ' ' + type + ' - footprint: ' + footprint + ' | moved: ' + piece.moved);
549         $(footprint).each(function() {
550             if (inside(this, position)) {
551                 dest = occupied('#' + this);
552                 if (type == 'knight') {
553                     // Knights are 'leapers'. They do not move along a vector 
554                     // but jump to the destination square. << B. Fisher
555                     if (dest.color == color) legal.push(dest);
556                     else legalIDs += writeID(this[0], this[1]);
557                 }
558                 else if (type == 'pawn') {
559                     // pawns cannot capture on their own column
560                     if (dest && C != this[0]) {
561                         if (dest.color == color) legal.push(dest);
562                         else legalIDs += writeID(this[0], this[1]);
563                     }
564                     // Check for a pawn in EP and whether its column matches the move
565                     else if (piece.EP && piece.EP.col() == this[0]) legalIDs += writeID(this[0], this[1]);
566                     // add vertical moves
567                     else if (C == this[0]) legalIDs += vector(position, this, color, false).list;
568                 }
569                 else {
570                     path = vector(position, this, color, true);
571                     legalIDs += path.list;
572                     legal.push(path.end);
573                 };
574             };
575         });
576         // King move legality.
577         if (type == 'king') {
578             // Check for castle legality and add king double step if true. << B. Fisher
579             if (piece.castle()) {
580                 var rook = occupied('#H' + rNum);
581                 // Kingside castle squares are unoccupied and unthreatened, and the kingside rook has not moved.
582                 if (vector(position, 'G' + rNum, color, false).list.match('G') && rook && !rook.moved && !check('F' + rNum, Game.Players[1])) {
583                     legalIDs += writeID('G', rNum);
584                 };
585                 // Queenside squares between rook and king are not occupied. The king is not moving across check.
586                 // and the queenside rook has not moved.
587                 var rook = occupied('#A' + rNum);
588                 if (vector(position, 'B' + rNum, color, false).list.match('B') && rook && !rook.moved && !check('D' + rNum, Game.Players[1])) {
589                     legalIDs += writeID('C', rNum);
590                 };
591             };
592             // Removes any square ids from legalIDs that would move the king into check << B. Fisher 3.04 1700
593             var squares = legalIDs.split(',');
594             $('.threat').removeClass('threat');
595             for (var i = squares.length - 2; i >= 0; i--) {
596                 if (check(squares[i], Game.Players[1])) {
597                     $(squares[i]).addClass('threat');
598                     squares.splice(i, 1);
599                 };
600             };
601             legalIDs = squares.join(',');
602         };
603         // === End King legality checks ===
604         legal['moves'] = legalIDs;
605         return legal;
606     };
607 /* Function: vector
608 	 * Peramiters: start - the pieces location, end - the final square along the given path, side - current players color
609 	 * 				capture - whether opposing pieces can be captured on the last square,
610 	 * 				pin - if true pass the first opponent piece on the vector
611 	 * Returns: a string of comma seperated square ids for the requested vector, continuing to an occupied square
612 	 * 			if the square is occupied by an opponent piece the vector ids includes that square unless capture is false.
613 	 * Coded by: B. Fisher
614 	 */

616     function vector(start, end, side, capture) {
617         var sX = cLabels.indexOf(start[0]),
618             eX = cLabels.indexOf(end[0]),
619             sY = start[1] * 1,
620             eY = end[1] * 1,
621             xInc = findInc(sX, eX),
622             yInc = findInc(sY, eY),
623             squareList = '',
624             dest, piece, square;
625         do {
626             sX += xInc;
627             sY += yInc;
628             square = cLabels[sX] + sY;
629             dest = occupied('#' + square);
630             if (dest) {
631                 if (capture && dest.color != side) squareList += '#' + square + ',';
632                 else if (dest.color == side) piece = dest;
633                 break;
634             };
635             squareList += '#' + square + ',';
636         }
637         while ((cLabels[sX] + sY) != end);
638         return {
639             list: squareList,
640             end: piece
641         };
642     };

679     function findInc(first, second) {
680         var Inc;
681         if (first > second) {
682             Inc = -1
683         }
684         else if (first < second) {
685             Inc = 1
686         }
687         else {
688             Inc = 0
689         };
690         return Inc;

696         var x = cLabels.indexOf(start[0]),
697             y = start[1] * 1;
698         do {
699             x += xInc;
700             y += yInc;
701         }
702         while (x > 0 && x < 7 && y > 1 && y < 8);
703         if (x < 0 || x > 7 || y < 1 || y > 8) {
704             return
705         };
706         return (cLabels[x] + y);

715     // Returns: pieces that are threatening the square location (requires string in 'RC' format where R = row and C = column).
716     // var ignore is an array of piece objects. It can be used to prevent recursion in the Legal object, helps in pinning checks.
717     // If threatening pieces are found return an array of their objects, else returns false. << B. Fisher

719     function check(square, player, ignore) {
720         var chk = new Array(),
721             ids, footprint;
722         $(player.pieces).each(function() {
723             if (this != ignore) {
724                 ids = Legal(this).moves;
725                 if (ids.length > 0 && ids.match(square)) chk.push(this);
726             };
727         });
728         // Evaluate player's pawn capture squares. << B. Fisher 3/14 2130
729         $(player.pawns).each(function() {
730             var pawn = this;
731             if (ignore != pawn) {
732                 ids = pawn.footprint();
733                 $(ids).each(function(index) {
734                     if (this[0] == pawn.col()) ids.splice(index, 1);
735                 });
736                 if ($.inArray(ids, square.substring(1)) >= 0) {
737                     chk.push(pawn);
738                 };
739             };
740         });
741         ids = player.King.footprint();
742         if ($.inArray(ids, square) >= 0) chk.push(player.King);
743         if (chk.length < 1) chk = false;
744         //	console.log('Player ' + player.color + ': ' + square + ' ' + chk);
745         return chk;
746     };
